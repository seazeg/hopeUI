import { Story, Canvas, Meta } from "@storybook/addon-docs/blocks";
import { useEffect } from "@storybook/client-api";
import { viewinfo } from "../view/lazyload.view.js";

# 图片懒加载（Lazyload）

---

<Meta title="复用组件/图片懒加载(Lazyload)" />

<Canvas>
    <Story
        name="图片懒加载(Lazyload)"
        height="150px"
        parameters={{
            docs: { source: { code: `${viewinfo.html}\n\n<script type="text/javascript">\n${viewinfo.js}\n</script>` } },
        }}
    >
        {() => {
            useEffect(() => {
                setTimeout(() => {
                    Function(viewinfo.js)();
                }, 100);
            });
            return viewinfo.html;
        }}
    </Story>
</Canvas>

## 基础属性

| 参数 | 说明          | 类型   | 可选值               | 默认值  |
| ---- | ------------- | ------ | -------------------- | ------- |
| ele  | 元素 dom 节点 | String | Class / id / TagName | TagName |

## 配置 options

| 参数       | 说明             | 类型    | 可选值 | 默认值 |
| ---------- | ---------------- | ------- | ------ | ------ |
| responsive | 是否是响应式图片 | Boolean | --     | true   |
| breakpoint | 自定义响应式断点 | Object  | --     | --     |
| adapimage  | 是否图片始终居中 | Boolean | --     | true   |
